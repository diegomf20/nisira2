SELECT TOP 1 IDESTADO 
FROM SECUENCIA_FLUJO  SF
INNER JOIN FORMULARIOS FS ON FS.FORMULARIO=SF.FORMULARIO_ORIGEN
WHERE SF.SECUENCIA<
	(	SELECT SECUENCIA 
		FROM SECUENCIA_FLUJO 
		WHERE IDESTADO='AP'
		AND FORMULARIO_ORIGEN=FS.FORMULARIO
	)
AND SUBSTRING(TABLA, 4, LEN ( TABLA )-1) = 'ORDENCOMPRA'
ORDER BY SECUENCIA DESC	
;
